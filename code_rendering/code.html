import functools
from typing import Any, Callable, Type, get_type_hints

def validate_types(func: Callable) -> Callable:
    """Decorator to validate function arguments against type hints."""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        hints = get_type_hints(func)
        # Map positional arguments to names
        all_args = kwargs.copy()
        all_args.update(zip(func.__code__.co_varnames, args))
        
        for name, value in all_args.items():
            if name in hints and not isinstance(value, hints[name]):
                raise TypeError(f"Argument '{name}' must be {hints[name]}")
        
        return func(*args, **kwargs)
    return wrapper

@validate_types
def process_transaction(user_id: int, amount: float, currency: str = "USD"):
    # Logic for processing financial data
    print(f"Processing {amount} {currency} for user {user_id}")
    return True

This pattern ensures that any input passed to process_transaction strictly adheres to the defined types at runtime, throwing a TypeError if a mismatch occurs.
